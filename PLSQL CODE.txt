PL/SQL CODE FOR BUTTONS:
START BUTTON :
BEGIN
	GO_BLOCK(‘EMP’);
	EXECUTE_QUERY;
END;
NEXT BUTTON:
BEGIN
	GO_BLOCK(‘EMP’);
	FIRST_RECORD;
END;
PREVIOUS BUTTON:
BEGIN
	GO_BLOCK(‘EMP’);
	PREVIOUS_RECORD;
END;

SAVE BUTTON:
BEGIN
	GO_BLOCK(‘EMP’);
	IF :ENAME IS NULL THEN
		MESSAGE(‘EMPLOYEE NAME MUST NOT BE EMPTY’);
	ELSE
		DO_KEY(‘CREATE_RECORD’);
		COMMIT;
	END IF;
END;


EXIT BUTTON:
BEGIN
	EXIT_FORM();
END;
